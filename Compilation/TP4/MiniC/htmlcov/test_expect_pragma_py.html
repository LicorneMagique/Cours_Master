33class TestExpectPragmas(object): 

34 """Base class for tests that read the expected result as annotations 

35 in test files. 

36 

37 get_expect(file) will parse the file, looking EXPECT and EXITCODE 

38 pragmas. 

39 

40 run_command(command) is a wrapper around subprocess.check_output() 

41 that extracts the output and exit code. 

42 

43 """ 

44 

45 def get_expect(self, filename): 

46 """Parse "filename" looking for EXPECT and EXITCODE annotations. 

47 

48 Look for a line "EXPECTED" (possibly with whitespaces and 

49 comments). Text after this "EXPECTED" line is the expected 

50 output. 

51 

52 The file may also contain a line like "EXITCODE <n>" where <n> 

53 is an integer, and is the expected exitcode of the command. 

54 

55 The result is cached to avoid re-parsing the file multiple 

56 times. 

57 """ 

58 if filename not in self.__expect: 

59 self.__expect[filename] = self._extract_expect(filename) 

60 return self.__expect[filename]

71 def run_command(self, cmd, scope="compile"): 

72 """Run the command cmd (given as [command, arg1, arg2, ...]), and 

73 return testinfo(exitcode=..., output=...) containing the 

74 exit code of the command it its standard output + standard error. 

75 

76 If scope="compile" (resp. "runtime"), then the exitcode (resp. 

77 execcode) is set with the exit status of the command, and the 

78 execcode (resp. exitcode) is set to 0. 

79 """ 

80 try: 

81 output = subprocess.check_output(cmd, timeout=60, 

82 stderr=subprocess.STDOUT) 

83 status = 0 

84 except subprocess.CalledProcessError as e: 

85 output = e.output 

86 status = e.returncode 

87 if scope == "runtime": 

88 return default_testinfo._replace(execcode=status, 

89 output=output.decode()) 

90 else: 

91 return default_testinfo._replace(exitcode=status, 

92 output=output.decode())

141 def _extract_expect(self, file): 

142 exitcode = 0 

143 execcode = 0 

144 linkargs = [] 

145 inside_expected = False 

146 skip_test_expected = False 

147 expected_lines = [] 

148 expected_present = False 

149 with open(file, encoding="utf-8") as f: 

150 for line in f.readlines(): 

151 # Ignore non-comments 

152 if not re.match(r'\s*//', line): 

153 continue 

154 # Cleanup comment start and whitespaces 

155 line = re.sub(r'\s*//\s*', '', line) 

156 line = re.sub(r'\s*$', '', line) 

157 

158 if line == 'END EXPECTED': 

159 inside_expected = False 

160 elif line.startswith('EXITCODE'): 

161 words = line.split(' ') 

162 assert len(words) == 2 

163 exitcode = int(words[1]) 

164 elif line.startswith('EXECCODE'): 

165 words = line.split(' ') 

166 assert len(words) == 2 

167 execcode = int(words[1]) 

168 elif line.startswith('LINKARGS'): 

169 words = line.split(' ') 

170 assert len(words) >= 2 

171 linkargs += [w.replace("$dir", os.path.dirname(file)) 

172 for w in words[1:]] 

173 elif line == 'EXPECTED': 

174 inside_expected = True 

175 expected_present = True 

176 elif line == 'SKIP TEST EXPECTED': 

177 skip_test_expected = True 

178 elif inside_expected: 

179 expected_lines.append(line) 

180 

181 if not expected_present: 

182 pytest.fail("Missing EXPECTED directive in test file") 

183 

184 if expected_lines == []: 

185 output = '' 

186 else: 

187 output = os.linesep.join(expected_lines) + os.linesep 

188 

189 return testinfo(exitcode=exitcode, execcode=execcode, 

190 output=output, linkargs=linkargs, 

191 skip_test_expected=skip_test_expected)
